import java.util.*;
class Main {
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int a[]=new int[n];
        for(int i=0;i<n;i++) a[i]=sc.nextInt();
        int key=sc.nextInt();
        boolean f=false;
        for(int i=0;i<n;i++) if(a[i]==key) f=true;
        System.out.println(f?"Found":"Not Found");
    }
}
