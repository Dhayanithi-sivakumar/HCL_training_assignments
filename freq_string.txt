import java.util.*;
class Main {
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        String s=sc.nextLine();
        char c=sc.next().charAt(0);
        int cnt=0;
        for(char x:s.toCharArray()) if(x==c) cnt++;
        System.out.println(cnt);
    }
}
