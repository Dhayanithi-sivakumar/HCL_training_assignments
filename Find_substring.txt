import java.util.*;
class Main {
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        String s=sc.nextLine();
        String sub=sc.nextLine();
        int cnt=0,i=0;
        while((i=s.indexOf(sub,i))!=-1){
            cnt++;
            i+=sub.length();
        }
        System.out.println(cnt);
    }
}
